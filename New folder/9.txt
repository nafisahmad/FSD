//welcome.js

const express = require("express");
const app = express();
app.get("/", (req, res) => {
  res.sendFile(__dirname + "/index.html");
});
app.get("/msg", (req, res) => {
  let pname = req.query.pname;

  res.send("<h1>Welcome " + pname + " " + "</h1>");
});
app.listen(3000, () => {
  console.log("Server started on port 3000...");
});
