      function validate() {
        var ids = new Array();
        ids[0] = "fn";
        ids[1] = "ln";
        ids[2] = "mail";
        ids[3] = "uid";
        ids[4] = "password";
        ids[5] = "confirmpassword";
        ids[6] = "mobile";
        var remarksids = new Array();
        remarksids[0] = "remarksfn";
        remarksids[1] = "remarksln";
        remarksids[2] = "remarksmail";
        remarksids[3] = "remarksuid";
        remarksids[4] = "remarkspassword";
        remarksids[5] = "remarksconfirmpassword";
        remarksids[6] = "remarksmobile";
        var remarks = new Array();
        remarks[0] = "<span>Please enter your first name</span>";
        remarks[1] = "<span>Please enter your last name</span>";
        remarks[2] = "<span>Please enter your e-mail address</span>";
        remarks[3] = "<span>Please enter your user ID</span>";

        remarks[4] = "<span>Please enter your password</span>";
        remarks[5] = "<span>Please enter your pasword and confirm it</span>";
        remarks[6] = "<span>Please enter a 10-digit valid mobile number</span>";
        remarks[7] = "<span>Passwords dont match!!!</span>";
        remarks[8] = "OK!";
        remarks[9] = "<span>Please enter a valid email id</span>";
        remarks[11] = "<span>Please enter a valid first name</span>";
        remarks[12] = "<span>Please enter a valid last name</span>";
        remarks[13] =
          "<span>Password should be min 3 chars and max 8 chars</span>";

        for (i in ids) {
          var userIp = document.getElementById(ids[i]).value;
          if (userIp == "") {
            document.getElementById(remarksids[i]).innerHTML = remarks[i];
          } else {
            if (i == 2) {
              var regularExpression =
                /^([a-zA-Z0-9-\._]+)@([a-zA-Z0-9]+)\.([a-z]+)(\.[a-z]+)?$/;
              if (regularExpression.test(userIp) == false) {
                document.getElementById(remarksids[i]).innerHTML = remarks[9];
              } else {
                document.getElementById(remarksids[i]).innerHTML = remarks[8];
              }
            } else if (i == 0 || i == 1) {
              var regularExpression = /^[a-zA-Z]+$/;
              if (regularExpression.test(userIp) == false) {
                if (i == 0) {
                  document.getElementById(remarksids[i]).innerHTML =
                    remarks[11];
                } else {
                  document.getElementById(remarksids[i]).innerHTML =
                    remarks[12];
                }
              } else {
                document.getElementById(remarksids[i]).innerHTML = remarks[8];
              }
            } else if (i == 4) {
              var n = userIp.length;
              if (n < 3 || n > 8) {
                document.getElementById(remarksids[i]).innerHTML = remarks[13];
              } else {
                document.getElementById(remarksids[i]).innerHTML = remarks[8];
              }
            } else if (i == 5) {
              var p = document.getElementById(ids[i - 1]).value;
              var cp = document.getElementById(ids[i]).value;
              if (p != cp) {
                document.getElementById(remarksids[i]).innerHTML = remarks[7];
              } else {
                document.getElementById(remarksids[i]).innerHTML = remarks[8];
              }
            } else if (i == 6) {
              var regularExpression = /^\d{10}$/;
              if (regularExpression.test(userIp) == false) {
                document.getElementById(remarksids[i]).innerHTML = remarks[6];
              } else {
                document.getElementById(remarksids[i]).innerHTML = remarks[8];
              }
            } else {
              document.getElementById(remarksids[i]).innerHTML = remarks[8];
            }
          }
        }
      }

      function finalValidate() {
        var remarksids = new Array();
        remarksids[0] = "remarksfn";
        remarksids[1] = "remarksln";
        remarksids[2] = "remarksmail";
        remarksids[3] = "remarksuid";
        remarksids[4] = "remarkspassword";
        remarksids[5] = "remarksconfirmpassword";
        remarksids[6] = "remarksmobile";
        var status = 1;
        for (i in remarksids) {
          if (document.getElementById(remarksids[i]).innerHTML != "OK!") {
            status = 0;
            break;
          }
        }
        if (status == 1) {
          document.getElementById("finalremarks").innerHTML =
            "All the details entered by you are correct!!!";
        }
      }
